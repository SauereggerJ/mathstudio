\title{Limit Superior and Limit Inferior}
\author{}
\date{}
\maketitle

Let $(x_n)$ be a sequence in $\mathbb{R}$.

\begin{align*}
Y_n &:= \sup_{k \geq n} x_k := \sup \{x_k : k \geq n\} \\
Z_n &:= \inf_{k \geq n} x_k := \inf \{x_k : k \geq n\}
\end{align*}

$Y_n$ and $Z_n$ are monotone sequences, so they converge in $\overline{\mathbb{R}} = \mathbb{R} \cup \{-\infty, \infty\}$.

\begin{align*}
\limsup_{n \to \infty} x_n &:= \lim_{n \to \infty} (\sup_{k \geq n} x_k) \quad \text{and} \\
\liminf_{n \to \infty} x_n &:= \lim_{n \to \infty} (\inf_{k \geq n} x_k)
\end{align*}

Or,

\begin{align*}
\limsup_{n \to \infty} x_n &= \inf_{n \in \mathbb{N}} (\sup_{k \geq n} x_k) \quad \text{and} \\
\liminf_{n \to \infty} x_n &= \sup_{n \in \mathbb{N}} (\inf_{k \geq n} x_k)
\end{align*}

\textbf{Notes:}
\begin{itemize}
    \item Any sequence $(x_n)$ in $\mathbb{R}$ has a smallest cluster point $x_*$ and a greatest cluster point $x^*$ in $\overline{\mathbb{R}}$ and these satisfy:
    \begin{align*}
        \liminf_{n \to \infty} x_n &= x_* \\
        \limsup_{n \to \infty} x_n &= x^*
    \end{align*}

    \item $(x_n)$ sequence in $\mathbb{R}$. Then
    
    $(x_n)$ converges in $\overline{\mathbb{R}}$ $\Leftrightarrow$ $\limsup_{n \to \infty} x_n \leq \liminf_{n \to \infty} x_n$
    
    $(x_n)$ converges in $\mathbb{R}$ $\Leftrightarrow$ $x = \lim_{n \to \infty} x_n = \limsup_{n \to \infty} x_n = \liminf_{n \to \infty} x_n$

    \item (Bolzano-Weierstrass):
    
    Every bounded sequence in $\mathbb{R}^m$ and $\mathbb{C}^m$ has a convergent subsequence, that is, a cluster point.
\end{itemize}
