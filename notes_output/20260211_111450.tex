\noindent\textbf{Theorem:} Let $X$ be a topological space and $Y$ a metric space. Let $f_n$ be a sequence of bounded functions of $X$ to $Y$, each of which is continuous. Let $f$ be a function from $X$ to $Y$ so that $\overline{\rho}(f_n, f) \to 0$ then $f$ is bounded and continuous.

\noindent\textbf{Proof:} Because $\overline{\rho}(f_n, f) \to 0$ there exists a $n\in\mathbb{N}$ such that $\overline{\rho}(f_n, f) < \frac{\varepsilon}{3}$. Pick an open set $A$ containing $x_0$, so $x\in A \implies \rho(f_n(x), f_n(x_0)) < \frac{\varepsilon}{3}$ which is possible because $f_n$ is continuous. For such $x$ we have
$$ \rho(f(x), f(x_0)) \le \rho(f(x), f_n(x)) + \rho(f_n(x), f_n(x_0)) + \rho(f_n(x_0), f(x_0)) $$
$$ < \frac{\varepsilon}{3} + \frac{\varepsilon}{3} + \frac{\varepsilon}{3} = \varepsilon $$
Thus $f$ is continuous at $x_0$.

For bounded observe $\exists n\in\mathbb{N}: \overline{\rho}(f_n, f) < 1$ and because $f_n$ is bounded: $\sup_{x\in X} \rho(f_n(x), y) < \infty$ and therefore:
$$ \sup_{x\in X} \rho(f(x), y) \le \sup_{x\in X} \rho(f_n(x), f(x)) + \sup_{x\in X} \rho(f_n(x), y) \le 1 + \infty$$

\noindent\textbf{Def:} $\rho(x, y)$ is a metric on $Y$, $f: X \to Y$ is bounded if and only if for one $y$ in $Y$: $\sup_{x\in X} (\rho(f(x), y)) < \infty$. Given two bounded functions from $X$ to $Y$ : $\overline{\rho}(f, g) := \sup_{x\in X} (\rho(f(x), g(x)))$.
