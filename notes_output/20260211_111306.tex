\section*{D: Product Metric}

\textbf{Definition:} Let $(X_j, d_j)$, $1 \le j \le m$ be metric spaces and $X := X_1 \times \cdots \times X_m$. Then the function
$$d(x, y) := \max_{1 \le j \le m} d_j(x_j, y_j)$$
where $x = (x_1, \ldots, x_m) \in X$ and $y = (y_1, \ldots, y_m) \in X$, is called the product metric. The metric space $X = (X, d)$ is called the product of the metric spaces $(X_1, d_1), \ldots, (X_m, d_m)$.

\textbf{Proposition:} For $a = (a_1, \ldots, a_m) \in X$ and $r > 0$, we have
$$\overline{B}_X(a, r) = \prod_{j=1}^{m} \overline{B}_{X_j}(a_j, r)$$

\textbf{Proof:} $x \in \overline{B}_X(a, r) \iff d(a, x) < r$
$$\iff x \in \left\{ x \in X \mid d(a, x) < r \right\}$$
$$\iff \max_{1 \le j \le m} d_j(a_j, x_j) < r$$
$$\iff d_j(a_j, x_j) < r \quad \forall j \in \{1, \ldots, m\}$$
$$\iff x_j \in \overline{B}_{X_j}(a_j, r) \quad \forall j \in \{1, \ldots, m\}$$
$$\iff x \in \prod_{j=1}^{m} \overline{B}_{X_j}(a_j, r)$$

\textbf{Proposition:} Let $X$ be the product of the metric spaces $(X_j, d_j)$, $1 \le j \le m$. Then the sequence $(x_n) = ((x_n^1, \ldots, x_n^m))_{n \in \mathbb{N}}$ converges in $X$ to the point $a = (a^1, \ldots, a^m)$ if and only if, for each $j \in \{1, \ldots, m\}$, the sequence $(x_n^j)_{n \in \mathbb{N}}$ converges in $X_j$ to $a^j \in X_j$.

\textbf{Proof:} \fbox{\ensuremath{\forall \epsilon > 0: \exists N_j \in \mathbb{N}: d_j(x_n^j, a^j) < \epsilon \quad \forall j \in \{1, \ldots, m\}, \text{ for } N := \max N_j \text{ we have } d(x_n, a) < \epsilon \text{ for all } n > N}}

On the other hand, if then for all $\epsilon > 0$ there exists a $N \in \mathbb{N}$ such that $d(x_n, a) = \max_{1 \le j \le m} d_j(x_n^j, a^j) < \epsilon$ for all $n > N$.
Then of course $d_j(x_n^j, a^j) < \epsilon$ for all $n > N$.
