\title{Finding a Subspace Complement for U in F^4}
\maketitle
\setcounter{enumi}{19}
\textbf{20)}\ \text{Suppose } \mathcal{U} = \{(x, x, y, y) \in \mathbb{F}^4 : x, y \in \mathbb{F}\}. \\ 
\text{Find a subspace } \mathcal{W} \text{ of } \mathbb{F}^4 \text{ such that } \mathbb{F}^4 = \mathcal{U} \oplus \mathcal{W}. \\ 
\textbf{Solution:}
\text{There are many possible solutions. This is one of them:}
$$\mathcal{W} = \{(x, 0, y, 0) \in \mathbb{F}^4 : x, y \in \mathbb{F}\}$$
\begin{enumerate}
    \item \(\mathcal{U} + \mathcal{W} = \mathbb{F}^4\)
    \text{Let } b \in \mathbb{F}^4, \text{ so } b = (b_1, b_2, b_3, b_4). \\ 
    \text{Choose } a \in \mathcal{U} \text{ with } a = (b_2, b_2, b_4, b_4) \text{ and } \\ 
    \overline{a} \in \mathcal{W} \text{ with } \overline{a} = (b_1 - b_2, 0, b_3 - b_4, 0), \text{ so } \\ 
    b = a + \overline{a}.
    \item \text{We show that the only possible way to write } 0 \text{ as a sum of } u \in \mathcal{U} \text{ and } w \in \mathcal{W} \text{ is to write } 0 = 0 + 0: \\ 
    \text{Let } u = (x_u, x_u, y_u, y_u) \in \mathcal{U} \text{ and } w = (x_w, 0, y_w, 0) \in \mathcal{W}. \\ 
    0 = u + w = (x_u + x_w, x_u, y_u + y_w, y_u).
    \text{This implies } x_u = 0 \text{ and } y_u = 0. \\ 
    \text{Let } x_u + x_w = 0 \implies 0 + x_w = 0 \implies x_w = 0 \\ 
    \text{and } y_u + y_w = 0 \implies 0 + y_w = 0 \implies y_w = 0. \\ 
    \text{Therefore } u = 0 \text{ and } w = 0. 
\end{enumerate}
\text{So } \mathcal{U} \oplus \mathcal{W} = \mathbb{F}^4.
