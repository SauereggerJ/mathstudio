\documentclass{article}\usepackage{amsmath, amssymb, amsthm}\title{Isomorphic Vector Spaces}\author{}\date{}\begin{document}\section*{Isomorphic Vector Spaces}\vspace{0.5em}\noindent\textbf{Definition:} An isomorphism is an invertible linear map.\begin{itemize}\item Two vector spaces are called isomorphic if there is an isomorphism from one vector space onto the other one.\end{itemize}\vspace{1em}\noindent\textbf{Theorem:} Two finite-dimensional vector spaces over a field $\mathbb{F}$ are isomorphic if and only if they have the same dimension.\vspace{1em}\noindent\textbf{Proof:}\begin{enumerate}\item First suppose $V$ and $W$ are isomorphic finite-dimensional vector spaces, and $T$ the isomorphism.\begin{itemize}\item Because $T$ is invertible, we have $\text{null}(T) = \{0\}$ and $\text{range}(T) = W$. Thus $\dim\left(\text{null } T\right) = 0$ and $\dim\left(\text{range } T\right) = \dim W$.\item Thus $\dim V = \dim\left(\text{null } T\right) + \dim\left(\text{range } T\right) = 0 + \dim W = \dim W$.\end{itemize}\item Suppose $V$ and $W$ are vector spaces with $\dim(V) = \dim(W) = n$. Let $\left\{v_1, \dots, v_n\right\}$ be a basis of $V$ and $\left\{w_1, \dots, w_n\right\}$ be a basis of $W$.\item Define $T \in \mathcal{L}(V, W)$ by:\begin{equation*}\label{eq:linear_map}\centering T\left(c_1 v_1 + \dots + c_n v_n\right) = c_1 w_1 + \dots + c_n w_n \end{equation*}\item Thus, $T$ is a well-defined isomorphism between $V$ and $W$.\end{enumerate}\n\section*{Recommended Reading}\n\begin{itemize}\n  \item \textbf{Groups, Matrices, and Vector Spaces} by James B.
