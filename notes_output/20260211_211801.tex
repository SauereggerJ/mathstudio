\begin{enumerate}
    \item Prove that the union of three subspaces $A, B, C$ of $V$ is a subspace of $V$ if and only if one of the subspaces contains the other two.
    \begin{itemize}
        \item $\Longleftarrow$ Easy: $A \cup B \cup C = A$ if $B \subseteq A$ and $C \subseteq A$, and $A$ is a subspace.
        \item $\Longrightarrow$ If $A \cup B \cup C$ is a subspace and there exists $x \in A$ but $x \notin B$ or $x \in A$ but $x \notin C$:
    \end{itemize}
    \item Prove that the union of two subspaces of $V$ is a subspace of $V$ if and only if one of the subspaces is contained in the other.
    \begin{enumerate}
        \item Proof: $1)$ $0$ has to be in $A$ and $B$, because both are subsets.
        \item $2)$ $a \in A \cup B$ implies $a \in A$ or $a \in B$.
        \item $\Longrightarrow 2)$ If none of the both is contained in the other then there exist two elements with the following conditions:
        $$a \in A \text{ and } a \notin B, \quad b \in B \text{ and } b \notin A.$$ 
        Because $A \cup B$ is a subspace: $a+b$ must be $\in A \cup B$
        $$\text{so } a+b \in A \text{ or } a+b \in B;$$ 
        but if $a+b \in A$ and $a \in A$ and so $-a \in A$:
        $$b = (a+b) + (-a) \text{ also must be element of } A. \\$$ 
        The same argument holds for $a+b \in B$, therefore.
        $A$ must be contained in $B$, or $B$ must be contained in $A$.
        $$\Longleftarrow \text{ If } A \subseteq B \text{ or } B \subseteq A \text{ then } A \cup B = A \text{ or } A \cup B = B.$$ 
    \end{enumerate}
\end{enumerate}
