\section*{Vector Space}

\textbf{Definition:}
Let $V$ be a set and $(F, +, \cdot)$ a field. Define the operations:
\begin{align*}
    + &: V \times V \to V \\
    \cdot &: F \times V \to V
\end{align*}
The set $V$ is a vector space over $F$ if:
\begin{itemize}
    \item $(V, +)$ is an abelian group.
    \item $\lambda \cdot (\mu \cdot w) = (\lambda \cdot \mu) \cdot w$ for all $\lambda, \mu \in F$ and $w \in V$.
    \item $\lambda \cdot (u + w) = \lambda u + \lambda w$ for all $\lambda \in F$ and $u, w \in V$.
    \item $(\lambda + \mu) \cdot w = \lambda w + \mu \cdot w$ for all $\lambda \in F$ and $u, w \in V$.
    \item $1 \cdot w = w$ for all $w \in V$, and $1$ is the multiplicative identity in $F$.
\end{itemize}

\section*{Metric Space}

\textbf{Definition:}
A set $X$, whose elements we shall call points, is said to be a metric space if for any two points $p$ and $q$ of $X$ are elements of the domain of a distance function $d(p, q): X \times X \to \mathbb{R}^+$ such that:
\begin{itemize}
    \item $d(p, q) > 0$ if $p \neq q$; $d(p, p) = 0$.
    \item $d(p, q) = d(q, p)$.
    \item $d(p, q) + d(q, r) \leq d(p, r)$ for any $r \in X$.
\end{itemize}
