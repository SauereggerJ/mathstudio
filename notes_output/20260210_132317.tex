\section*{Metric Spaces}
Let $X$ be a set. A function $d: X\times X \to \mathbb{R}_0^+$ is called
a metric on $X$, if:
\begin{enumerate}
    \item[M1] $d(x,y) = 0 \iff x=y$
    \item[M2] $d(x,y) = d(y,x)$, \quad for all $x,y\in X$
    \item[M3] $d(x,y) \le d(x,z) + d(z,y)$, \quad for all $x,y,z\in X$
\end{enumerate}
\textbf{Notes:}
\begin{enumerate}
    \item The open ball is $B(a,r) := \{x \in X : d(a,x) < r \}$.
    \item The closed ball is $\overline{B}(a,r) := \{x \in X : d(a,x) \le r \}$.
    \item Let $(X_j, d_j)$, $1\le j \le m$, be metric spaces, and
    $X := X_1 \times \dots \times X_m$. Then
    $$d(x,y) := \max_{1\le j \le m} d_j(x_j, y_j)$$
is a metric on $X$, and
    $$B_X(a,r) = \prod_{j=1}^{m} B_{X_j}(a_j, r),\quad \overline{B}_X(a,r) = \prod_{j=1}^{m} \overline{B}_{X_j}(a_j, r)$$
    \item For all $x,y,z \in X$ of a metric space:
    $$d(x,y) \ge |d(x,z) - d(z,y)|$$
\end{enumerate}
